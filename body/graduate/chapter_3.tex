\chapter{在线稀疏高斯过程回归}

\section{稀疏高斯过程及其变分推断}

标准高斯过程回归在训练集规模为 $N$ 时，需要对 $N \times N$ 的核矩阵进行 Cholesky
分解，时间复杂度为 $\mathcal{O}(N^3)$，存储复杂度为 $\mathcal{O}(N^2)$。这一瓶颈在
自动驾驶等实时系统中尤为突出——车辆运行过程中持续产生传感器数据，直接应用标准 GPR
难以满足在线更新的延时约束。稀疏高斯过程（Sparse Gaussian Process，SGP）通过引入一组
数量远少于训练集的\textbf{诱导点}（inducing points）来近似完整的后验分布，将计算复杂度
降低至可接受的范围。变分推断为稀疏高斯过程提供了严谨的理论基础。其核心思想是引入一个变分分
布 $q(u)$ 来近似诱导点函数值 $u$ 的真实后验，并通过最大化证据下界（ELBO）来优化该分布。
这一框架不仅将计算复杂度降至 $O(NM^2)$（$M$为诱导点数量），还能够提供在线学习的可能性。

\subsection{稀疏高斯过程的两个条件分布}

设诱导输入集合 $Z = [z_1, z_2, \cdots, z_M]^\top \in \mathbb{R}^{M \times D}$，
$M \ll N$，对应的诱导变量为
\begin{equation}
    u = f(Z) = [f(z_1), f(z_2), \cdots, f(z_M)]^\top
\end{equation}
由高斯过程的定义，$(f(X), u)$，$(f(x^*), u)$ 服从联合高斯分布：
\begin{equation}
    \begin{pmatrix} f(X) \\ u \end{pmatrix} \sim \mathcal{N}
    \left(0, \begin{bmatrix} K_{ff} & K_{fu} \\ K_{uf} & K_{uu} \end{bmatrix}\right), \quad 
    \begin{pmatrix} f(x^*) \\ u \end{pmatrix} \sim \mathcal{N}
    \left(0, \begin{bmatrix} K_{**} & K_{*u} \\ K_{u*} & K_{uu} \end{bmatrix}\right)
\end{equation}
基于高斯过程的一致性，我们可以通过积分（边缘化）$u$联合分布$p(f^*, f, u)$恢复$p(f^*, f)$：
\begin{equation}
    p(f^*, f) = \int p(f^*, f, u) \mathrm{d}u, \quad p(u) = \mathcal N(0, K_{uu})
\end{equation}
这是精确的表达式，接下来我们介绍几乎所有稀疏方法采取的近似假设：\textbf{给定$u$的条件下$f$和$f^*$是条件独立的}。
\begin{equation}
    p(f^*, f) \approxeq q(^*, f) = \int q(f^* | u) q(f | u) p(u) \mathrm{d}u
\end{equation}
“诱导变量”这一名称的由来是：由于$f$和 $f^*$只能通过$u$进行信息传递，因此$u$起到了诱导训练案例与测试案例之间依赖关系的作用。


利用条件高斯公式，训练函数值和测试函数值在给定诱导变量 $u$ 条件下的精确分布为
\begin{equation} \label{eq:conditionals}
\begin{aligned}
    \text{训练条件（training conditional）}: &\quad p({f}|{u}) = \mathcal N \big (K_{fu} K_{uu}^{-1} u, 
    K_{ff} - Q_{ff} \big ) \\
    \text{测试条件（test conditional）}: &\quad p({f}_*|{u}) = \mathcal N \big (K_{*u} K_{uu}^{-1} u, 
    K_{**} - Q_{**} \big )
\end{aligned}
\end{equation}
其中 $Q_{ab} = K_{au} K_{uu}^{-1} K_{ub}$ 称为 Nyström 近似矩阵，刻画了由诱导点
所能解释的函数值方差。$K_{ff} - Q_{ff}$ 为残差协方差，反映了诱导点无法捕捉的
函数变化量。

大部分的稀疏方法都可以看作是对这两个条件分布的不同近似方法，我们接下来便引入DTC和FITC。

\subsection{DTC与FITC近似}

\textbf{DTC}\cite{csato2002sparse}基于$f = K_{fu} K_{uu}^{-1} u$这一映射，将两个条件分布定义为：
\begin{equation}
    q_{\mathrm{DTC}}(f|u)=\mathcal{N}(K_{fu}K_{uu}^{-1}{u},0),
    \quad q_{\mathrm{DTC}}({f_{*}}|{u})=p({f_{*}}|{u})
\end{equation}
DTC的近似之处在于其训练条件分布，给定$u$下，训练数据$f$是确定的，而其测试条件没有经过近似，是精确的。

其预测分布为
\begin{equation} \label{equ:DTC_pred}
    \begin{aligned}
    q_{\text{DTC}}(f_* | y) &= \mathcal N \left (
        Q_{*f} (Q_{ff} + \sigma^2 I)^{-1} y,
        K_{**} - Q_{*f} (Q_{ff} + \sigma^2 I)^{-1} Q_{f*}
        \right ) \\
        &= \mathcal N \left (
            \sigma^{-2} K_{*u} \Sigma K_{uf} y,
            K_{**} - Q_{**} + K_{*u} \Sigma K_{u, *}
            \right )
    \end{aligned}
\end{equation}
其中
\begin{equation}
    \Sigma = (\sigma^{-2} K_{uf} K_{fu} + K_{uu})^{-1}
\end{equation}
此处我们使用了逆矩阵公式，将需要取逆的矩阵的维度从$\mathbb R^{N \times N}$降为了$\mathbb R^{M \times M}$：
\begin{equation}
    \begin{aligned}
        (Q_{ff} + \sigma^2 I)^{-1} 
        &= (K_{{fu}} K_{{uu}}^{-1} K_{{uf}} + \sigma^2 I)^{-1} \\
        &\rightarrow \text{使用逆矩阵公式} \\
        &= \sigma^{-2} I - \sigma^{-4} K_{{fu}} (K_{{uu}} + \sigma^{-2} K_{{uf}} K_{{fu}})^{-1} K_{{uf}} \\
        &= \sigma^{-2} I - \sigma^{-4} K_{{fu}} \Sigma K_{{uf}} \\
        &\rightarrow \sigma^{-2} K_{{uf}} K_{{fu}} = \Sigma^{-1} - K_{{uu}} \\
        Q_{*f} (Q_{ff} + \sigma^2 I)^{-1}
        &= \sigma^{-2} K_{{*u}} K_{{uu}}^{-1} K_{{uf}} - \sigma^{-2} K_{{*u}} K_{{uu}}^{-1} (\Sigma^{-1} - K_{{uu}}) \Sigma K_{{uf}} \\
        &= \sigma^{-2} K_{{*u}} \Sigma K_{{uf}}
    \end{aligned}
\end{equation}

\textbf{FITC}\cite{snelson2007local}则假设给定$u$的条件下，$f_i$彼此之间相互独立
\begin{equation}
    p(y | f) \simeq q(y | u) = \mathcal N \left ( 
        K_{fu} K_{uu}^{-1} u, 
        \mathrm{diag} [K_{ff} - Q_{ff}] + \sigma^2 I
         \right )
\end{equation}
那么两个条件分布为：
\begin{equation}
    q_{\mathrm{FITC}}({f|u})
    = \prod_{i=1}^{n} p(f_i | u)
    =\mathcal{N}(K_{{fu}}K_{{uu}}^{-1}{u}, \mathrm{diag} [K_{ff} - Q_{ff}]),
    \quad q_{\mathrm{FITC}}({f_{*}}|{\bf u})=p({f_{*}}|{\bf u})
\end{equation}

FITC并非直接忽略方差，而是通过引入额外的独立性假设来近似训练条件分布。其先验形式为（这里我们暂时只考虑单个测试样本的情况）：
\begin{equation}
    q_{\text{FITC}}(f, f_*) = \mathcal N \left ( 0, \begin{bmatrix}
        Q_{ff} + \mathrm{diag} [K_{ff} - Q_{ff}] & Q_{f*} \\
        Q_{*f} & K_{**}
    \end{bmatrix} \right )
\end{equation}

最终我们可以得到其预测公式：
\begin{equation}
    \begin{aligned}
    q_{\text{FITC}}(f_* | y) &= \mathcal N \left (
        Q_{*f} (Q_{ff} + \Lambda)^{-1} y,
        K_{**} - Q_{*f} (Q_{ff} + \Lambda)^{-1} Q_{f*}
        \right ) \\
        &= \mathcal N \left (
            K_{*u} \Sigma K_{uf} \Lambda^{-1} y,
            K_{**} - Q_{**} + K_{*u} \Sigma K_{u, *}
            \right )
    \end{aligned}
\end{equation}
其中$\Sigma = (K_{uf} \Lambda^{-1} K_{fu} + K_{uu})^{-1}$， 
$\Lambda = \mathrm{diag} [K_{ff} - Q_{ff} + \sigma^2 I]$.

从DTC到FITC的演变，体现了稀疏高斯过程近似中对不确定性建模的深化——DTC以确定性映射为代价追求简洁，
FITC则通过条件独立假设保留了局部不确定性，从而在保持计算效率的同时提升了模型的泛化能力。

\subsection{高斯过程的变分推断}
上文基于直接的假设推导了DTC和FITC的预测公式，这缺乏一定的理论保证，因此我们从变分推断的角度，推导出DTC的
公式（FITC同理），并基于此我们可以设计在线更新算法。

设变分分布$q(u) = \mathcal N(m_u, S_u)$，变分后验为
\begin{equation}
    q(f, u) = p(f | u) q(u)
\end{equation}

在变分推断的框架下，我们需要最大化对数边缘似然$\log p(y)$，并由此引入证据下界和$q(u)$和$p(u)$之间的
KL散度。
\begin{equation}
    \begin{aligned}
        \log p(y) &= \log \iint p(y, f, u) df du \\
        &= \log \iint p(y, f, u) \frac{q(f, u)}{q(f, u)} df du \\
        &= \log \iint q(f, u) \frac{p(y, f, u)}{q(f, u)} df du \\
        &= \log \mathbb E_{q(f, u)} \left [ \frac{p(y, f, u)}{q(f, u)} \right ] \\
        &\geq \mathbb E_{q(f, u)} \left [ \log \frac{p(y, f, u)}{q(f, u)} \right ] \quad \text{(Jensen不等式)} \\
        &= \mathbb E_{q(f, u)} \left [ \log \frac{p(y|f, u)p(f, u)}{p(f|u)q(u)} \right ] \\
        &= \mathbb E_{q(f, u)} \left [ \log \frac{p(y|f)p(f|u)p(u)}{p(f|u)q(u)} \right ] \\
        &= \mathbb E_{q(f, u)} \left [ \log \frac{p(y|f)p(u)}{q(u)} \right ] \\
        &= \mathbb E_{q(f, u)} \left [ \log p(y|f) \right ] - \mathbb E_{q(f, u)} \left [ \log \frac{q(u)}{p(u)} \right ] \\
        &= \mathbb E_{q(f, u)} \left [ \log p(y|f) \right ] - \text{KL} \left [q(u) || p(u) \right ]
    \end{aligned}
\end{equation}
显然，最大化边缘似然就是最小化$q(u)$和$p(u)$之间的KL散度。

其中证据变分下界ELBO就是：
\begin{equation}
    \mathcal L(q) = \mathbb E_{q(f, u)} \left [ \log \frac{p(y, f, u)}{q(f, u)} \right ]
    = E_{q(f, u)} \left [ \log p(y|f) \right ] - \text{KL} \left [q(u) || p(u) \right ]
\end{equation}

我们考虑第一项
\begin{equation}
    \begin{aligned}
        \mathbb E_{q(f, u)} \left [ \log p(y|f) \right ] &= \int q(f, u) \log p(y|f) df du \\
        &= \int p(f|u) q(u) \log p(y|f) df du \\
        &= \int q(u) \int P(f|u) \log p(y|f) df du
    \end{aligned}
\end{equation}
其中
\begin{equation}
    \begin{aligned}
        y|f & \sim \mathcal N(f, \sigma^2) \\
        f|u & \sim \mathcal N(K_{fu}K_{uu}^{-1}u, K_{ff} - Q_{ff})
    \end{aligned}
\end{equation}
那么
\begin{equation}
    \log p(y|f) = -\frac{N}{2} \log 2\pi - N \log \sigma - \frac{1}{2\sigma^2} (y-f)^T(y-f)
\end{equation}
我们可以得到内部的积分
\begin{equation}
    \begin{aligned}
        \int p(f|u) \log p(y|f) df &= -\frac{N}{2} \log 2\pi - N \log \sigma - \frac{1}{2\sigma^2} \mathbb E_{p(f|u)} \left [ (y-f)^T(y-f) \right ] \\
        & \text{使用二次型的期望公式} \\
        &= -\frac{N}{2} \log 2\pi - N \log \sigma \\
        &- \frac{1}{2\sigma^2} \left [ (y - K_{fu}K_{uu}^{-1}u)^T(y - K_{fu}K_{uu}^{-1}u) + \text{Tr}(K_{ff} - Q_{ff}) \right ] \\
    \end{aligned}
\end{equation}
这样总的积分就是
\begin{equation}
    \begin{aligned}
        \mathbb E_{q(f, u)} \left [ \log p(y|f) \right ] &= -\frac{N}{2} \log 2\pi - N \log \sigma \\
        &- \frac{1}{2\sigma^2} \left [ (y - K_{fu}K_{uu}^{-1}m_u)^T(y - K_{fu}K_{uu}^{-1}m_u) + \text{Tr}(K_{ff} - Q_{ff}) \right ] \\
        &- \frac{1}{2\sigma^2} \text{Tr}(K_{fu}K_{uu}^{-1}S_u K_{uu}^{-1}K_{uf})
    \end{aligned}
\end{equation}

我们接着考虑第二项KL散度， $p(u) \sim \mathcal N(0, K_{uu})$， $q(u) \sim \mathcal N(m_u, S_u)$：
\begin{equation}
    \begin{aligned}
        -\text{KL} \left [q(u) || p(u) \right ] &=
        -\mathbb E_{q(f, u)} \left [ \log q(u) \right ] + \mathbb E_{q(f, u)} \left [ \log p(u) \right ] \\
        &= -\mathbb E_{q(u)} \left [ \log q(u) \right ] + \mathbb E_{q(u)} \left [ \log p(u) \right ]
    \end{aligned}
\end{equation}
其中
\begin{equation}
    \begin{aligned}
        \mathbb E_{q(u)} \left [ \log q(u) \right ] &= -\frac{M}{2} \log 2 \pi - \frac{1}{2} \log | S_u |
        -\frac{1}{2} \mathbb E_{q(u)} \left [ (u-m_u)^\top S_u^{-1} (u-m_u) \right ] \\
        &= -\frac{M}{2} \log 2 \pi - \frac{1}{2} \log | S_u | - \frac{M}{2}
    \end{aligned}  
\end{equation}
并且
\begin{equation}
    \begin{aligned}
        \mathbb E_{q(u)} \left [ \log p(u) \right ] &= -\frac{M}{2} \log 2 \pi - \frac{1}{2} \log | K_{uu} |
        -\frac{1}{2} \mathbb E_{q(u)} \left [ u^\top K_{uu}^{-1} u \right ] \\
        &= -\frac{M}{2} \log 2 \pi - \frac{1}{2} \log | K_{uu} | \\
        &-\frac{1}{2} \mathbb E_{q(u)} \left [ (m_u + (u-m_u))^\top K_{uu}^{-1} (m_u + (u-m_u)) \right ] \\
        &= -\frac{M}{2} \log 2 \pi - \frac{1}{2} \log | K_{uu} | - \frac{1}{2} m_u^T K_{uu}^{-1} m_u
        -\frac{1}{2} \text{Tr}(K_{uu}^{-1}S_u)
    \end{aligned}  
\end{equation}
那么
\begin{equation}
    \begin{aligned}
        -\text{KL} \left [q(u) || p(u) \right ] &=
        -\frac{1}{2} \left [ -\log |S_u| - M + \log |K_{uu}| + m_u^T K_{uu}^{-1} m_u + \text{Tr}(K_{uu}^{-1}S_u) \right ]
    \end{aligned}
\end{equation}

只保留关于$m_u$和$S_u$的项，ELBO的剩余的表达式是
\begin{equation}
    \begin{aligned}
        \tilde{\mathcal L(q)} &= \frac{1}{2} \left [
            \log |S_u| - \text{Tr}(K_{uu}^{-1}S_u) - m_u^T K_{uu}^{-1} m_u
        \right ] \\
        & -\frac{1}{2 \sigma^2} \left [
            -2y^\top K_{fu} K_{uu}^{-1} m_u + m_u^\top K_{uu}^{-1} K_{uf} K_{fu} K_{uu}^{-1} m_u
            + \text{Tr}(K_{fu}K_{uu}^{-1}S_u K_{uu}^{-1}K_{uf})
        \right ]
    \end{aligned}
\end{equation}

分别对$m_u$和$S_u$求导数：
\begin{equation}
    \begin{aligned}
        \frac{\partial \tilde{\mathcal L(q)}}{\partial m_u} &= 
        -K_{uu}^{-1} m_u + \sigma^{-2} K_{uu}^{-1} K_{uf} y - \sigma^{-2} 
        K_{uu}^{-1} K_{uf} K_{fu} K_{uu}^{-1} m_u \\
        \frac{\partial \tilde{\mathcal L(q)}}{\partial S_u} &= 
        \frac{1}{2} S_u^{-1} - \frac{1}{2} K_{uu}^{-1} - \frac{1}{2 \sigma^2} 
        K_{uu}^{-1} K_{uf} K_{fu} K_{uu}^{-1} \\
    \end{aligned}
\end{equation}
分别等于0得到最优值
\begin{equation}
    \begin{aligned}
        m_u &= \sigma^{-2} K_{uu} \Sigma K_{uf} y \\
        S_u &= K_{uu} \Sigma K_{uu}
    \end{aligned}
\end{equation}
其中
\begin{equation}
    \Sigma = (K_{uf} K_{fu} / \sigma^2 + K_{uu})^{-1}
\end{equation}

根据总期望公式和总方差公式我们可以得到：
\begin{equation}
    \begin{aligned}
    \mathbb E(f_* | y) &= \mathbb E_{q(u)} [\mathbb E(f_* | u)] = \mathbb E_{q(u)} [K_{*u} K_{uu}^{-1} u] = K_{*u} K_{uu}^{-1} m_u \\
    \text{Var}(f_* | y) &= \mathbb E_{q(u)} [\text{Var}(f_* | u)] + \text{Var}_{q(u)} [\mathbb E(f_* | u)] \\
    &= \mathbb E_{q(u)} [K_{**} - Q_{**}] + \text{Var}_{q(u)} [K_{*u} K_{uu}^{-1} u] \\
    &= K_{**} - Q_{**} + K_{*u} K_{uu}^{-1} S_u K_{uu}^{-1} K_{u*}
\end{aligned}
\end{equation}
我们便可以得到预测分布
\begin{equation} \label{equ:mu_su_predictive}
    \begin{aligned}
        q(f_* | y) &= \int p(f_* | u) q(u) du \\
        &= \mathcal N (K_{*u} K_{uu}^{-1} m_u, K_{**} - Q_{**} + K_{*u} K_{uu}^{-1} S_u K_{uu}^{-1} K_{u*}) \\
        &= \mathcal N (\sigma^{-2} K_{*u} \Sigma K_{uf} y, K_{**} - Q_{**} + K_{*u} \Sigma K_{u*})
    \end{aligned}
\end{equation}

这跟我们推导的DTC的\autoref{equ:DTC_pred}一模一样，但是我们得到了ELBO得到的理论下界保证。同理在FITC的框架下，
诱导变量的变分分布是
\begin{equation}
    \begin{aligned}
        m_u &= K_{uu} \Sigma K_{uf} \Lambda^{-1} y \\
        S_u &= K_{uu} \Sigma K_{uu}
    \end{aligned}
\end{equation}
其中
\begin{equation}
    \Sigma = (K_{uf} \Lambda^{-1} K_{fu} + K_{uu})^{-1}, \quad \Lambda = \text{diag}(K_{ff} - Q_{ff} + \sigma^2)
\end{equation}

\section{在线更新算法}

在许多实际应用场景中，数据通常以流式（streaming）的形式不断到达，传统的批处理（batch）推断方法需要反复在全量数据
上进行优化，计算代价较高且难以满足实时性需求。因此，有必要设计能够随数据逐步更新的在线推断算法。

本节围绕变分推断框架，讨论模型在新数据到达时的更新策略，根据是否需要扩充诱导变量及其分布分为非增量和增量更新算法。

\subsection{变分分布的非增量更新}

在$k-1$时刻，我们有变分分布$q(u) = \mathcal N (m_{k-1}, S_{k-1})$，在$k$时刻我们得到了一个新的数据点
$(x_k, y_k)$，$p(y_k | u) = \mathcal N(K_{f_k u} K_{uu}^{-1}u, \sigma^2)$，$u$在$y_k$下的条件概率（后验）为
\begin{equation}
    q(u|y_k) = \frac{q(y_k|u)q(u)}{p(y_k)} \propto q(y_k|u)q(u)
\end{equation}
取负对数
\begin{equation}
    \begin{aligned}
        - \log q(y_k|u)q(u)
        &= \frac{1}{2} (u - m_{k-1})^T S_{k-1}^{-1} (u - m_{k-1}) + \frac{1}{2\sigma^2} (y_k - K_{f_ku} K_{uu}^{-1} u)^2 + \text{常量} \\
        &\rightarrow \text{只保留依赖于} u \text{的项} \\
        &= \frac{1}{2} u^T (S_{k-1}^{-1} + \sigma^{-2} K_{u u}^{-1} K_{uf_k} K_{f_ku} K_{uu}^{-1}) u \\
        &- (S_{k-1}^{-1} m_{k-1} + \sigma^{-2} K_{uu}^{-1} K_{uf_k} y_k) u \\
        &= \frac{1}{2} u^T A u - b^T u + \text{常量} \\
        &= \frac{1}{2} (u - A^{-1} b)^T A (u - A^{-1} b) + \text{常量}
    \end{aligned}
\end{equation}
这个后验是一个指数二次型，这必然是一个高斯分布，于是我们得到后验分布$q(u|y_k) = \mathcal N(m_k, S_k)$：
\begin{equation}    
    \begin{aligned}
        S_k &= A^{-1} =  (S_{k-1}^{-1} + \sigma^{-2} K_{u u}^{-1} K_{u f_k} K_{f_k u} K_{u u}^{-1})^{-1} \\
        m_k &= A^{-1}b = S_k (S_{k-1}^{-1} m_{k-1} + \sigma^{-2} K_{u u}^{-1} K_{u f_k} y_k)
    \end{aligned}   
\end{equation}
我们可以使用逆矩阵定理处理这个逆：
\begin{equation}
    S_k = S_{k-1} - S_{k-1} K_{u u}^{-1} K_{u f_k} (\sigma^2 + K_{f_k u} K_{u u}^{-1} S_{k-1} K_{u u}^{-1} K_{u f_k})^{-1} K_{f_k u} K_{u u}^{-1} S_{k-1}
\end{equation}

这是一个贝叶斯更新步骤，我们可以将整个更新方案写成一个类似卡尔曼滤波的形式，定义$\phi_k = K_{f_k u} K_{uu}^{-1}$，那么
$y_k = \phi_k u + \epsilon, \quad \epsilon \sim \mathcal N(0, \sigma^2)$
\begin{equation}
    \begin{aligned}
        & r_k = y_k - \Phi_k m_{k-1} \\
        & s_k = \sigma^2 + \Phi_k S_{k-1} \Phi_k^T \\
        & K_k = S_{k-1} \Phi_k^T s_k^{-1} \\
        & m_k = m_{k-1} + K_k r_k \\
        & S_k = S_{k-1} - K_k \phi_k S_{k-1}
    \end{aligned}
\end{equation}
第一项是残差，第二项是预测方差，第三项是增益。总结一下即为\autoref{alg:online_vsgp_update}。

\begin{algorithm}[htbp]
    \caption{在线变分稀疏高斯过程更新}
    \label{alg:online_vsgp_update}
    \begin{algorithmic}[1]
        \Require 
        \Statex 上一时刻变分分布参数：均值 $m_{k-1}$，协方差 $S_{k-1}$
        \Statex 诱导点协方差矩阵 $K_{uu}$（固定）
        \Statex 新观测数据点 $(x_k, y_k)$
        \Statex 观测噪声方差 $\sigma^2$
        \Statex 核函数 $k(\cdot, \cdot)$
        
        \Function{OnlineUpdate}{$m_{k-1}$, $S_{k-1}$, $K_{uu}$, $x_k$, $y_k$, $\sigma^2$}
        
            \State \textbf{步骤1}: 计算核向量
            \State $K_{f_k u} \gets [k(x_k, u_1), k(x_k, u_2), \ldots, k(x_k, u_M)]$ \Comment{$M$为诱导点个数}
            
            \State \textbf{步骤2}: 计算线性化观测矩阵
            \State $\phi_k \gets K_{f_k u} K_{uu}^{-1}$ \Comment{将观测映射到诱导点空间}
            
            \State \textbf{步骤3}: 计算预测
            \State $\hat{y}_k \gets \phi_k m_{k-1}$ \Comment{预测观测值}
            \State $r_k \gets y_k - \hat{y}_k$ \Comment{残差（新息）}
            
            \State \textbf{步骤4}: 计算预测方差
            \State $s_k \gets \sigma^2 + \phi_k S_{k-1} \phi_k^\top$ \Comment{标量预测方差}
            
            \State \textbf{步骤5}: 计算增益
            \State $K_k \gets S_{k-1} \phi_k^\top s_k^{-1}$ \Comment{增益向量}
            
            \State \textbf{步骤6}: 更新后验分布
            \State $m_k \gets m_{k-1} + K_k r_k$ \Comment{更新均值}
            \State $S_k \gets S_{k-1} - K_k \phi_k S_{k-1}$ \Comment{更新协方差}
            
            \State \textbf{return} $m_k$, $S_k$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\subsection{变分分布的增量更新}

在标准的变分稀疏高斯过程回归中，若仅对变分分布进行非增量更新，则诱导点的位置保持不变，仅对其对应的变分均值与协方差进行调整。
在这种情况下，模型的表达能力受限于已有诱导点在输入空间中的分布范围。

当测试输入 $x^*$ 位于训练数据及诱导点未充分覆盖的区域时，由于缺乏有效的支撑信息，高斯过程的后验预测将逐渐退化为其先验分布，
即回归到先验均值 $m(x^*)$，同时伴随较大的不确定性。这种现象表明，固定诱导点位置的变分更新机制在面对输入分布扩展或新区域数据时，
难以提供准确的预测结果。

为了解决上述问题，有必要对变分分布进行增量扩展，在保留已有统计信息的基础上，引入新的诱导点以覆盖新的输入区域，从而提升模型在未
见区域的泛化能力与预测精度。

观察\autoref{equ:mu_su_predictive}我们发现，增加诱导变量会导致标准稀疏高斯过程预测分布的$K_{u u}$， $K_{u f}$这两项变化，
需要更新，从变分推断的角度则需要更新$K_{u u}$，$m_u$，$S_u$。

设新诱导点为$u_{n}$，原有诱导点为$u_{o}$，$K_{u f}$的更新比较简单：
\begin{equation}
    K_{u f} = \begin{bmatrix}
        K_{u_o f} \\
        K_{u_n f}
    \end{bmatrix}
\end{equation}

$K_{u u}$则并不会直接更新，而是更新对应的Cholesky分解矩阵$L_{u u}$，令
\begin{equation}
    C = L_{u_o u_o}^{-1} K_{u_o u_n}, \quad D = \sqrt{K_{u_n u_n} - C^\top C}
\end{equation}
则
\begin{equation}
    L_{u u} = \begin{bmatrix}
        L_{u_o u_o} & 0 \\
        C^\top & D
    \end{bmatrix}
\end{equation}

$m_u$的更新方法为
\begin{equation}
    m_u = \begin{bmatrix}
        m_{u_o} \\
        (L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n})^\top m_{u_o}
    \end{bmatrix}
\end{equation}

$S_u$的更新方法为
\begin{equation}
    S_u = \begin{bmatrix}
        S_{u_o} & (S_{u_o} L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n})^\top \\
        S_{u_o} L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n} & S_{n n}
    \end{bmatrix}
\end{equation}
其中
\begin{equation}
    S_{n n} = K_{u_n u_n} - (L_{u_o u_o}^{-1} K_{u_o u_n})^\top (L_{u_o u_o}^{-1} K_{u_o u_n}) + 
    (L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n})^\top S_{u_o} (L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n})
\end{equation}

这是由$u_n = K_{u_n u_o} K_{u_o u_o}^{-1} S_{u_o}$得来的，我们将上述增量更新方法记为\autoref{alg:add_new_inducing}。

\begin{algorithm}[htbp]
    \caption{诱导点增量更新}
    \label{alg:add_new_inducing}
    \begin{algorithmic}[1]
        \Require 
        \Statex 新诱导变量$u_o$及其协方差矩阵的分解矩阵$L_{u_o u_o}$
        \Statex 新旧诱导点间的协方差$K_{u_o u_n}$，新诱导点自身的协方差$K_{u_n u_n}$
        \Statex 旧变分分布参数：均值 $m_{u_o}$，协方差 $S_{u_o}$

        \Function{AddNewInducing}{$u_n$, $L_{u_o u_o}$, $K_{u_o u_n}$, $K_{u_n u_n}$, $m_{u_o}$, $S_{u_o}$}
            
            \State \textbf{步骤1}: 计算中间矩阵C和D
            \State $C \gets L_{u_o u_o}^{-1} K_{u_o u_n}$
            \State $D \gets \sqrt{K_{u_n u_n} - C^\top C}$
            
            \State \textbf{步骤2}: 更新Cholesky因子$L_{uu}$
            \State $L_{uu} \gets \begin{bmatrix} L_{u_o u_o} & 0 \\ C^\top & D \end{bmatrix}$
            
            \State \textbf{步骤3}: 更新均值$m_u$
            \State $T \gets L_{u_o u_o}^{-\top} L_{u_o u_o}^{-1} K_{u_o u_n}$ \Comment{临时变量}
            \State $m_u \gets \begin{bmatrix} m_{u_o} \\ T^\top m_{u_o} \end{bmatrix}$
            
            \State \textbf{步骤4}: 更新协方差$S_u$的右下块$S_{nn}$
            \State $A \gets L_{u_o u_o}^{-1} K_{u_o u_n}$ \Comment{临时变量}
            \State $S_{nn} \gets K_{u_n u_n} - A^\top A + T^\top S_{u_o} T$
            
            \State \textbf{步骤5}: 组装完整的$S_u$矩阵
            \State $S_{u_o u_n} \gets S_{u_o} T$
            \State $S_u \gets \begin{bmatrix} S_{u_o} & S_{u_o u_n}^\top \\ S_{u_o u_n} & S_{nn} \end{bmatrix}$
            
            \State \textbf{return} $L_{uu}$, $m_u$, $S_u$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
